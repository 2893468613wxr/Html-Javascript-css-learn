JS类型转换

JavaScript 类型转换是指将一种数据类型转换为另一种数据类型。

JavaScript 类型转换有以下几种：

  • String(字符串)
  • Number(数字)
  • Boolean(布尔值)
  • Array(数组)
  • Object(对象)
  • Function(函数)

以下是 JavaScript 类型转换的示例：

String(字符串)

将数字转换为字符串：

        var num = 10;
        var str = num.toString();
        console.log(str); // "10"

将布尔值转换为字符串：

        var bool = true;
        var str = bool.toString();
        console.log(str); // "true"

将数组转换为字符串：

        var arr = [1, 2, 3];
        var str = arr.toString();
        console.log(str); // "1,2,3"

将对象转换为字符串：

        var obj = {name: "John", age: 30};
        var str = obj.toString();
        console.log(str); // "[object Object]"

将函数转换为字符串：

        function myFunction() {
            console.log("Hello World!");
        }
        var str = myFunction.toString();
        console.log(str); // "function myFunction() { console.log("Hello World!"); }"

Number(数字)

将字符串转换为数字：

        var str = "10";
        var num = Number(str);
        console.log(num); // 10

将布尔值转换为数字：

        var bool = true;
        var num = Number(bool);
        console.log(num); // 1

将数组转换为数字：

        var arr = [1, 2, 3];
        var num = Number(arr);
        console.log(num); // NaN

将对象转换为数字：

        var obj = {name: "John", age: 30};
        var num = Number(obj);
        console.log(num); // NaN

将函数转换为数字：

        function myFunction() {
            console.log("Hello World!");
        }
        var num = Number(myFunction);
        console.log(num); // NaN

Boolean(布尔值)

将字符串转换为布尔值：

        var str = "true";
        var bool = Boolean(str);
        console.log(bool); // true

将数字转换为布尔值：

        var num = 10;
        var bool = Boolean(num);
        console.log(bool); // true

将数组转换为布尔值：

        var arr = [1, 2, 3];
        var bool = Boolean(arr);
        console.log(bool); // true

将对象转换为布尔值：

        var obj = {name: "John", age: 30};
        var bool = Boolean(obj);
        console.log(bool); // true

将函数转换为布尔值：

        function myFunction() {
            console.log("Hello World!");
        }
        var bool = Boolean(myFunction);
        console.log(bool); // true

Array(数组)

将字符串转换为数组：

        var str = "1,2,3";
        var arr = str.split(",");
        console.log(arr); // ["1", "2", "3"]

将数字转换为数组：

        var num = 10;
        var arr = [num];
        console.log(arr); // [10]

将布尔值转换为数组：

        var bool = true;
        var arr = [bool];
        console.log(arr); // [true]

将对象转换为数组：

        var obj = {name: "John", age: 30};
        var arr = [obj];
        console.log(arr); // [{name: "John", age: 30}]

将函数转换为数组：

        function myFunction() {
            console.log("Hello World!");
        }
        var arr = [myFunction];
        console.log(arr); // [function myFunction() { console.log("Hello World!"); }]

Object(对象)

将字符串转换为对象：

        var str = "name=John&age=30";
        var obj = {};
        var arr = str.split("&");
        for (var i = 0; i < arr.length; i++) {
            var temp = arr[i].split("=");
            obj[temp[0]] = temp[1];
        }
        console.log(obj); // {name: "John", age: "30"}

将数字转换为对象：

        var num = 10;
        var obj = {num: num};
        console.log(obj); // {num: 10}

将布尔值转换为对象：

        var bool = true;
        var obj = {bool: bool};
        console.log(obj); // {bool: true}

将数组转换为对象：

        var arr = [1, 2, 3];
        var obj = {arr: arr};
        console.log(obj); // {arr: Array(3)}

将函数转换为对象：

        function myFunction() {
            console.log("Hello World!");
        }
        var obj = {myFunction: myFunction};
        console.log(obj); // {myFunction: function myFunction() { console.log("Hello World!"); }}

Function(函数)

将字符串转换为函数：

        var str = "function myFunction() { console.log('Hello World!'); }";
        var func = new Function(str);
        func(); // "Hello World!"

将数字转换为函数：

        var num = 10;
        var func = new Function("num", "console.log(num)");
        func(num); // 10

将布尔值转换为函数：

        var bool = true;
        var func = new Function("bool", "console.log(bool)");
        func(bool); // true

将数组转换为函数：

        var arr = [1, 2, 3];
        var func = new Function("arr", "console.log(arr)");
        func(arr); // [1, 2, 3]

将对象转换为函数：

        var obj = {name: "John", age: 30};
        var func = new Function("obj", "console.log(obj)");
        func(obj); // {name: "John", age: 30}

将函数转换为函数：

        function myFunction() {
            console.log("Hello World!");
        }
        var func = new Function("func", "func()");
        func(myFunction); // "Hello World!"
